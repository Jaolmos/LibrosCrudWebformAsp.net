CREATE TABLE Tipo (
    TipoID INT IDENTITY(1,1) PRIMARY KEY,
    Descripcion VARCHAR(50) NOT NULL
);

CREATE TABLE Tematica (
    TematicaID INT IDENTITY(1,1) PRIMARY KEY,
    Descripcion VARCHAR(100) NOT NULL
);

CREATE TABLE Libros (
    LibroID INT IDENTITY(1,1) PRIMARY KEY,
    Titulo VARCHAR(255) NOT NULL,
    Autor VARCHAR(255) NOT NULL,
    ISBN VARCHAR(13) NOT NULL,
    TipoID INT,
    TematicaID INT,
    FOREIGN KEY (TipoID) REFERENCES Tipo(TipoID),
    FOREIGN KEY (TematicaID) REFERENCES Tematica(TematicaID)
);
